/**
 * Codex Signum — BOCPD Detector
 *
 * Per-metric Bayesian Online Change Point Detector using a
 * Normal-Inverse-Gamma conjugate prior.
 *
 * Each detector instance is bound to a single metric stream.
 * Callers supply and receive back a BOCPDState so that state can be
 * persisted externally (e.g. on the Bloom morpheme's `bocpdState` property).
 *
 * @stub — NIG message-passing implementation pending t2
 * @module codex-signum-core/signals/BOCPDDetector
 */

import type { BOCPDConfig, BOCPDSignal, BOCPDState } from "./types.js";

/** Return value from a single BOCPD update step. */
export interface BOCPDUpdateResult {
  /** Signal emitted for this observation */
  signal: BOCPDSignal;
  /** Updated NIG state to persist for the next call */
  nextState: BOCPDState;
}

/**
 * Bayesian Online Change Point Detector for a single metric stream.
 *
 * Instantiate one detector per metric via BOCPDRegistry.
 * The detector is stateless between calls — pass the last BOCPDState in
 * and receive the next one back from `update()`.
 */
export class BOCPDDetector {
  /** Immutable configuration for this detector instance */
  readonly config: Readonly<BOCPDConfig>;

  constructor(config: BOCPDConfig) {
    this.config = Object.freeze({ ...config });
  }

  /**
   * Initialise a fresh BOCPDState from this detector's prior configuration.
   *
   * Call once to obtain the seed state before the first `update()` invocation.
   */
  initialState(): BOCPDState {
    return {
      mu0: this.config.mu0,
      kappa: this.config.kappa,
      alpha: this.config.alpha,
      beta: this.config.beta,
      runLengths: [1.0], // single hypothesis: run length 0 with full mass
    };

import { BOCPDDetector } from "./BOCPDDetector.js";
import type { BOCPDConfig, BOCPDSignal, BOCPDState } from "./types.js";

/**
 * Registry that owns one BOCPDDetector per metric ID.
 *
 * The registry is the single source of truth for per-metric BOCPD state.
 * Consumers read back the updated BOCPDState via `getState()` and may
 * persist it onto the corresponding Bloom morpheme.
 */
export class BOCPDRegistry {
  /** Default config used when `getOrCreate` is called without an explicit config */
  private readonly defaultConfig: Readonly<BOCPDConfig>;

  private readonly detectors = new Map<string, BOCPDDetector>();
  private readonly states = new Map<string, BOCPDState>();

  constructor(defaultConfig: BOCPDConfig) {
    this.defaultConfig = Object.freeze({ ...defaultConfig });
  }

  /**
   * Return the detector for `metricId`, creating it with `config` (or the
   * registry's default config) if it does not yet exist.
   */
  getOrCreate(metricId: string, config?: BOCPDConfig): BOCPDDetector {
    let detector = this.detectors.get(metricId);
    if (!detector) {
      detector = new BOCPDDetector(config ?? this.defaultConfig);
      this.detectors.set(metricId, detector);
      this.states.set(metricId, detector.initialState());
    }
    return detector;
  }
