export interface TaskOutputLinkageReport {
  totalTaskOutputs: number;
  linkedToDispatch: {
    succeeded: number;
    failed: number;
    total: number;
  };
  orphaned: {
    succeeded: number;
    failed: number;
    total: number;
  };
  linkageRate: number;
}

// Add this new section after ANALYTICS QUERIES (around line 820):

// ============ TASK OUTPUT VERIFICATION QUERIES ============

/**
 * Verify that all TaskOutput nodes link to DISPATCH Resonator regardless of
 * success or failure status. Used for M-9.VA-FIX verification to ensure the
 * DISPATCH stage receives all task outputs for final processing.
 *
 * Orphaned TaskOutput nodes indicate a failure in the task output linkage
 * mechanism and must be zero post-fix.
 */
export async function verifyTaskOutputLinkage(): Promise<TaskOutputLinkageReport> {
  const result = await runQuery(
    `MATCH (to:TaskOutput)
     OPTIONAL MATCH (r:Resonator {stage: 'DISPATCH'})-[:PROCESSED]->(to)
     WITH to,
          r IS NOT NULL AS isLinked,
          to.status AS status
     RETURN count(*) AS totalCount,
            sum(CASE WHEN isLinked AND status = 'succeeded' THEN 1 ELSE 0 END) AS linkedSucceeded,
            sum(CASE WHEN isLinked AND status = 'failed' THEN 1 ELSE 0 END) AS linkedFailed,
            sum(CASE WHEN isLinked THEN 1 ELSE 0 END) AS linkedTotal,
            sum(CASE WHEN NOT isLinked AND status = 'succeeded' THEN 1 ELSE 0 END) AS orphanedSucceeded,
            sum(CASE WHEN NOT isLinked AND status = 'failed' THEN 1 ELSE 0 END) AS orphanedFailed,
            sum(CASE WHEN NOT isLinked THEN 1 ELSE 0 END) AS orphanedTotal`,
    {},
    "READ",
  );

  if (result.records.length === 0) {
    return {
      totalTaskOutputs: 0,
      linkedToDispatch: {
        succeeded: 0,
        failed: 0,
        total: 0,
      },
      orphaned: {
        succeeded: 0,
        failed: 0,
        total: 0,
      },
      linkageRate: 0,
    };
  }

  const rec = result.records[0];
  const totalCount = Number(rec.get("totalCount") ?? 0);
  const linkedSucceeded = Number(rec.get("linkedSucceeded") ?? 0);
  const linkedFailed = Number(rec.get("linkedFailed") ?? 0);
  const linkedTotal = Number(rec.get("linkedTotal") ?? 0);
  const orphanedSucceeded = Number(rec.get("orphanedSucceeded") ?? 0);
  const orphanedFailed = Number(rec.get("orphanedFailed") ?? 0);
  const orphanedTotal = Number(rec.get("orphanedTotal") ?? 0);

  const linkageRate = totalCount > 0 ? linkedTotal / totalCount : 0;

  return {
    totalTaskOutputs: totalCount,
    linkedToDispatch: {
      succeeded: linkedSucceeded,
      failed: linkedFailed,
      total: linkedTotal,
    },
    orphaned: {
      succeeded: orphanedSucceeded,
      failed: orphanedFailed,
      total: orphanedTotal,
    },
    linkageRate,
  };
}

interface CheckResult {
  name: string;
  status: "PASS" | "FAIL" | "INFO" | "DEFERRED";
  detail: string;

const results: CheckResult[] = [];

function pass(name: string, detail: string) {

function deferred(name: string, detail: string) {

async function verify() {

    const bloom = first.get("bloom");

        "PipelineRun nodes",
        `${prResult.records.length} run(s), latest: status=${status}, tasks=${tasks}, quality=${quality}`,

  const toQuery = targetRunId

       WHERE pr.id = $runId
       RETURN pr.id AS runId, count(to) AS taskOutputs,
              avg(to.qualityScore) AS avgQuality,
              collect(DISTINCT to.modelUsed) AS models`
    : `MATCH (pr:PipelineRun)-[:PRODUCED]->(to:TaskOutput)

        `${provenance.missingBoth} decisions missing provenance`,

      `    - Succeeded: ${linkageReport.linkedToDispatch.succeeded}`,
