// ============ MODEL ENDPOINT VERIFICATION QUERIES ============

/**
 * Verify that no retired model endpoints are referenced by ModelArm selections.
 * Used for Thompson sampling router validation to prevent 404 errors from
 * retired endpoints like claude-opus-4-1.
 */
export async function verifyModelEndpoints(): Promise<ModelEndpointVerification> {
  const result = await runQuery(
    `MATCH (s:Seed)
     RETURN s.id AS seedId,
            s.modelId AS modelId,
            s.endpoint AS endpoint,
            s.status AS status,
            s.provider AS provider
     ORDER BY s.status DESC, s.modelId ASC`,
    {},
    "READ",
  );

  const activeEndpoints: ModelEndpointVerification["activeEndpoints"] = [];
  const retiredEndpoints: ModelEndpointVerification["retiredEndpoints"] = [];
  const retiredModelIds = new Set<string>();

  // Known retired model IDs that should not appear in active endpoints
  const knownRetiredModels = new Set(["claude-opus-4-1", "gpt-4-turbo-preview"]);

  for (const rec of result.records) {
    const seedId = String(rec.get("seedId") ?? "");
    const modelId = String(rec.get("modelId") ?? "");
    const endpoint = rec.get("endpoint");
    const status = String(rec.get("status") ?? "");
    const provider = rec.get("provider");

    const endpointStr = endpoint ? String(endpoint) : undefined;

    if (status === "retired" || knownRetiredModels.has(modelId)) {
      retiredEndpoints.push({
        seedId,
        modelId,
        endpoint: endpointStr,
        status,
      });
      retiredModelIds.add(modelId);
    } else if (status === "active") {
      activeEndpoints.push({
        seedId,
        modelId,
        endpoint: endpointStr || "",
        status,
        provider: provider ? String(provider) : undefined,
      });
    }
  }

  return {
    totalSeeds: result.records.length,
    activeEndpoints,
    retiredEndpoints,
    retiredModelIds: Array.from(retiredModelIds),
  };
}

    qualityScore: 0.85,
    durationMs: 1500,

  );

      "\n❌ ERROR: Retired model references found in graph:",

      "\nThompson sampling router would hit 404 errors from these endpoints.",

    console.error(
      "Please remove retired models before proceeding with routing.",

      `  ✅ ${endpoint.modelId}`,

    console.log(
      `     Seed ID: ${endpoint.seedId}`,

    if (endpoint.endpoint) {
      console.log(`     Endpoint: ${endpoint.endpoint}`);

    if (endpoint.provider) {
      console.log(`     Provider: ${endpoint.provider}`);

    "MATCH (d:Decision {id: $id}) DETACH DELETE d",

  );

  );
