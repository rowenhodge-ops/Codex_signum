export interface DecisionProvenanceReport {
  totalDecisions: number;
  withRunId: number;
  withoutRunId: number;
  withTaskId: number;
  withoutTaskId: number;
  missingBoth: number;
  completedWithoutProvenance: number;

}

// ============ PROVENANCE VERIFICATION QUERIES ============

/**
 * Verify Decision node provenance chains include required runId and taskId fields.
 * Used for M-9.VA-FIX verification to ensure all Decisions can receive human feedback.
 *
 * Decision nodes without runId/taskId are permanently disconnected from the human
 * feedback loop, preventing Thompson sampling posteriors from incorporating human signal.
 */
export async function verifyDecisionProvenance(): Promise<DecisionProvenanceReport> {
  const result = await runQuery(
    `MATCH (d:Decision)
     WITH d,
          d.runId IS NOT NULL AS hasRunId,
          d.taskId IS NOT NULL AS hasTaskId
     WITH count(*) AS totalCount,
          sum(CASE WHEN hasRunId THEN 1 ELSE 0 END) AS withRunId,
          sum(CASE WHEN NOT hasRunId THEN 1 ELSE 0 END) AS withoutRunId,
          sum(CASE WHEN hasTaskId THEN 1 ELSE 0 END) AS withTaskId,
          sum(CASE WHEN NOT hasTaskId THEN 1 ELSE 0 END) AS withoutTaskId,
          sum(CASE WHEN NOT hasRunId AND NOT hasTaskId THEN 1 ELSE 0 END) AS missingBoth,
          sum(CASE WHEN d.status = 'completed' AND NOT hasRunId THEN 1 ELSE 0 END) AS completedWithoutProvenance,
          collect(CASE WHEN (NOT hasRunId OR NOT hasTaskId) THEN d.id ELSE null END) AS orphanIds
     RETURN totalCount, withRunId, withoutRunId, withTaskId, withoutTaskId, 
            missingBoth, completedWithoutProvenance, orphanIds`,
    {},
    "READ",
  );

  if (result.records.length === 0) {
    return {
      totalDecisions: 0,
      withRunId: 0,
      withoutRunId: 0,
      withTaskId: 0,
      withoutTaskId: 0,
      missingBoth: 0,
      completedWithoutProvenance: 0,
      orphanDecisionIds: [],
    };
  }

  const rec = result.records[0];
  const orphanIds = rec.get("orphanIds") ?? [];
  
  return {
    totalDecisions: Number(rec.get("totalCount") ?? 0),
    withRunId: Number(rec.get("withRunId") ?? 0),
    withoutRunId: Number(rec.get("withoutRunId") ?? 0),
    withTaskId: Number(rec.get("withTaskId") ?? 0),
    withoutTaskId: Number(rec.get("withoutTaskId") ?? 0),
    missingBoth: Number(rec.get("missingBoth") ?? 0),
    completedWithoutProvenance: Number(rec.get("completedWithoutProvenance") ?? 0),
    orphanDecisionIds: orphanIds
      .filter((id: string | null) => id !== null)
      .slice(0, 100) as string[], // Cap at 100 IDs to avoid huge outputs
  };
}

interface CheckResult {
  name: string;
  status: "PASS" | "FAIL" | "INFO" | "DEFERRED";

const results: CheckResult[] = [];

}

  results.push({ name, status: "FAIL", detail });
  console.log(`  ❌ [FAIL] ${name}: ${detail}`);
}

function info(name: string, detail: string) {
  results.push({ name, status: "INFO", detail });

}

function deferred(name: string, detail: string) {

// ── Verification checks ─────────────────────────────────────────────────────

async function verify() {
  console.log("══════════════════════════════════════════════════════════════");
  console.log("  M-9.VA Graph Verification");
  if (targetRunId) {
    console.log(`  Target run: ${targetRunId}`);
  } else {
    console.log("  Target run: most recent");
  }
  console.log("══════════════════════════════════════════════════════════════\n");

  // ── Check 0: Constitutional rules (baseline) ────────────────────────────

  console.log("── Check 0: Constitutional Rules (baseline) ──");
  const rules = await runQuery(
    "MATCH (r:ConstitutionalRule) RETURN r.name AS name, r.status AS status",
    {},
    "READ",
  );
  if (rules.records.length >= 10) {
    pass("Constitutional rules", `${rules.records.length} rules found`);
  } else {
    fail("Constitutional rules", `Expected ≥10, found ${rules.records.length}`);

       WHERE pr.id = $runId
       OPTIONAL MATCH (b:Bloom) WHERE b.id = pr.bloomId

              pr.taskCount AS tasks, pr.overallQuality AS quality,
              coalesce(b.name, pr.bloomId) AS bloom
       ORDER BY pr.startedAt DESC LIMIT 5`
    : `MATCH (pr:PipelineRun)
       OPTIONAL MATCH (b:Bloom) WHERE b.id = pr.bloomId
       RETURN pr.id AS runId, pr.intent AS intent, pr.status AS status,
              pr.taskCount AS tasks, pr.overallQuality AS quality,
              coalesce(b.name, pr.bloomId) AS bloom
       ORDER BY pr.startedAt DESC LIMIT 5`;
  const prResult = await runQuery(prQuery, targetRunId ? { runId: targetRunId } : {}, "READ");

  if (prResult.records.length === 0) {
    fail("PipelineRun nodes", "No PipelineRun nodes found");
  } else {
    const first = prResult.records[0];
    const status = first.get("status");
    const tasks = first.get("tasks");
    const quality = first.get("quality");
    const bloom = first.get("bloom");
    const runId = first.get("runId");

    console.log(`  Status: ${status}, Tasks: ${tasks}, Quality: ${quality}, Bloom: ${bloom}`);

    if (status === "completed" && tasks > 0) {
      pass("PipelineRun nodes", `${prResult.records.length} run(s), latest: status=${status}, tasks=${tasks}, quality=${quality}`);
    } else {
      fail("PipelineRun nodes", `Latest run: status=${status}, tasks=${tasks} (expected completed with tasks > 0)`);
    }
  }

  // ── Check 2: TaskOutput nodes with correct relationships ────────────────

  console.log("\n── Check 2: TaskOutput nodes ──");
  const toQuery = targetRunId

    : `MATCH (pr:PipelineRun)-[:PRODUCED]->(to:TaskOutput)
       WHERE pr.status = 'completed'
       WITH pr, to ORDER BY pr.startedAt DESC
       WITH pr, count(to) AS taskOutputs,
              avg(to.qualityScore) AS avgQuality,
              collect(DISTINCT to.modelUsed) AS models
       LIMIT 1
       RETURN pr.id AS runId, taskOutputs, avgQuality, models`;

    const count = rec.get("taskOutputs");

    } else {

  const resQuery = targetRunId

    } else {
      fail("Decision provenance", `${provenance.missingBoth} Decisions missing both runId/taskId, ${provenance.completedWithoutProvenance} completed without provenance`);
