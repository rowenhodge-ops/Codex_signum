/**
 * Get quality score distribution statistics grouped by task type.
 * Used for M-9.VA-FIX verification to confirm genuine variance exists.
 */
export async function getQualityScoreDistribution(): Promise<QualityScoreDistribution[]> {
  const result = await runQuery(
    `MATCH (to:TaskOutput)
     WHERE to.qualityScore IS NOT NULL
     RETURN to.taskType AS taskType,
            count(to) AS count,
            avg(to.qualityScore) AS mean,
            stDevP(to.qualityScore) AS stddev,
            min(to.qualityScore) AS min,
            max(to.qualityScore) AS max
     ORDER BY taskType`,
    {},
    "READ",
  );

  return result.records.map((rec) => ({
    taskType: String(rec.get("taskType") ?? "unknown"),
    count: Number(rec.get("count") ?? 0),
    mean: Number(rec.get("mean") ?? 0),
    stddev: Number(rec.get("stddev") ?? 0),
    min: Number(rec.get("min") ?? 0),
    max: Number(rec.get("max") ?? 0),
  }));
}
