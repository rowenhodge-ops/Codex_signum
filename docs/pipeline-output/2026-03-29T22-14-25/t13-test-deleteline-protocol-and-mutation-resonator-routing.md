/**
 * Creates a new Morpheme node in the graph, enforcing Highlander Protocol.
 *
 * @param morphemeType - The type of morpheme to instantiate (e.g., 'seed', 'bloom').
 * @param properties - The properties for the new morpheme. Must include 'id', 'name', 'status'.
 * @param parentId - The ID of the parent Bloom node.
 * @param highlanderOptions - Options for Highlander Protocol enforcement (e.g., definition IDs).
 * @param labels - Additional labels to apply to the morpheme.
 * @returns A Result object indicating success or failure, with the ID of the new morpheme if successful.
 */
export async function instantiateMorpheme(
  morphemeType: MorphemeType,
  properties: MorphemeProperties,
  parentId: string,
  highlanderOptions?: HighlanderOptions,
  labels?: string[]
): Promise<Result> {
  assert(morphemeType, "morphemeType is required");
  assert(properties.id, "properties.id is required");
  assert(parentId, "parentId is required");

  // Type check and property validation
  const morphemeLabels = new MorphemeLabels(morphemeType, labels);
  const typeCheckResult = typeCheckMorphemeProperties(morphemeType, properties, morphemeLabels);
  if (!typeCheckResult.success) {
    return typeCheckResult;
  }

  // Highlander Protocol validation
  const validationResult = await validateMorphemeAgainstHighlander(
    morphemeType,
    properties,
    parentId,
    highlanderOptions
  );
  if (!validationResult.success) {
    return validationResult;
  }

  const timestamp = new Date().toISOString();
  const allLabels = morphemeLabels.toArray().join(":");
  const initialProperties = {
    ...properties,
    createdAt: timestamp,
    updatedAt: timestamp,
  };

  try {
    const result = await writeTransaction(async (tx) => {
      const query = `
        MATCH (parent:Bloom {id: $parentId})
        CREATE (n:${allLabels} $initialProperties)
        CREATE (parent)-[:CONTAINS {createdAt: $timestamp}]->(n)
        ${highlanderOptions?.definitionId ? `CREATE (n)-[:INSTANTIATES {createdAt: $timestamp}]->(:ConstitutionalBloom:def {id: $highlanderDefinitionId})` : ""}
        RETURN n.id AS morphemeId, n.name AS name
      `;
      const params = {
        parentId,
        initialProperties,
        timestamp,
        highlanderDefinitionId: highlanderOptions?.definitionId,
      };
      const writeResult = await tx.run(query, params);

      if (writeResult.records.length === 0) {
        throw new Error("Failed to create morpheme or link to parent.");
      }

      return {
        success: true,
        morphemeId: writeResult.records[0].get("morphemeId"),
        name: writeResult.records[0].get("name"),
      };
    });
    return result;
  } catch (error) {
    console.error(`Error instantiating morpheme ${properties.id}:`, error);
    return { success: false, error: `Failed to instantiate morpheme: ${error.message}` };
  }
}

/**
 * Updates an existing Morpheme node.
 * Properties are updated, new labels can be added from an allowlist.
 *
 * @param morphemeId - The ID of the morpheme to update.
 * @param properties - An object containing properties to update.
 * @param addLabels - Optional array of labels to add. Allowed labels are ['Archived'].
 * @returns A Result object indicating success or failure.
 */
export async function updateMorpheme(
  morphemeId: string,
  properties: Partial<MorphemeProperties>,
  addLabels?: string[]
): Promise<Result> {
  assert(morphemeId, "morphemeId is required for updateMorpheme");
  assert(properties || addLabels, "Either properties or addLabels must be provided for updateMorpheme");

  if (addLabels) {
    const allowedLabels = ['Archived'];
    const invalidLabels = addLabels.filter(label => !allowedLabels.includes(label));
    if (invalidLabels.length > 0) {
      return { success: false, error: `Attempted to add disallowed labels: ${invalidLabels.join(', ')}` };
    }
  }

  const timestamp = new Date().toISOString();
  const propertiesToUpdate: Record<string, any> = { ...properties, updatedAt: timestamp };

  // Prepare parameters for SET clause
  const setClauses: string[] = [];
  const updateParams: Record<string, any> = { morphemeId };

  for (const [key, value] of Object.entries(propertiesToUpdate)) {
    setClauses.push(`n.${key} = $upd_${key}`);
    updateParams[`upd_${key}`] = value;
  }

  const addLabelClause = addLabels && addLabels.length > 0 ? `SET n:${addLabels.join(':')}` : '';

  try {
    const result = await writeTransaction(async (tx) => {
      const query = `
        MATCH (n {id: $morphemeId})
        ${setClauses.length > 0 ? `SET ${setClauses.join(', ')}` : ''}
        ${addLabelClause}
        RETURN n.id AS morphemeId, labels(n) AS newLabels, properties(n) AS newProperties
      `;
      const writeResult = await tx.run(query, updateParams);

      if (writeResult.records.length === 0) {
        return { success: false, error: `Morpheme with ID ${morphemeId} not found.` };
      }

      const record = writeResult.records[0];
      return {
        success: true,
        morphemeId: record.get("morphemeId"),
        newLabels: record.get("newLabels"),
        newProperties: record.get("newProperties"),
      };
    });
    return result;
  } catch (error) {
    console.error(`Error updating morpheme ${morphemeId}:`, error);
    return { success: false, error: `Failed to update morpheme: ${error.message}` };
  }
}

/**
 * Creates a new Line node connecting two existing Morpheme nodes.
 *
 * @param lineId - The ID for the new line.
 * @param lineType - The type of line to create (e.g., 'FLOWS_TO').
 * @param fromId - The ID of the source Morpheme.
 * @param toId - The ID of the target Morpheme.
 * @param properties - Additional properties for the line relationship.
 * @returns A Result object indicating success or failure.
 */
export async function createLine(
  lineId: string,
  lineType: LineType,
  fromId: string,
  toId: string,
  properties?: LineProperties
): Promise<Result> {
  assert(lineId, "lineId is required");
  assert(lineType, "lineType is required");
  assert(fromId, "fromId is required");
  assert(toId, "toId is required");

  const timestamp = new Date().toISOString();
  const initialProperties = {
    ...properties,
    id: lineId, // Ensure the line node itself has an ID property
    createdAt: timestamp,
    updatedAt: timestamp,
    status: "active", // Lines are active by default
  };

  try {
    const result = await writeTransaction(async (tx) => {
      // 1. Validate existence of fromId and toId
      const checkNodesQuery = `
        MATCH (from {id: $fromId})
        MATCH (to {id: $toId})
        RETURN count(from) AS fromCount, count(to) AS toCount
      `;
      const checkResult = await tx.run(checkNodesQuery, { fromId, toId });
      const record = checkResult.records[0];

      if (record.get("fromCount") === 0) {
        return { success: false, error: `Source node with ID ${fromId} not found.` };
      }
      if (record.get("toCount") === 0) {
        return { success: false, error: `Target node with ID ${toId} not found.` };
      }

      // 2. Create the Line node and relationship
      // Note: A "Line" is a node, and the relationship is usually an additional semantic link.
      // For simplicity, this implementation assumes the "Line" is purely the relationship.
      // If "Line" is a distinct node, an adjustment would be needed.
      // Based on common patterns in similar systems, "Line" can be a relationship.
      // If it's a node, then we'd CREATE (l:Line {id: $lineId, ...}) and then (from)-[:FLOWS_TO]->(l)-[:POINTS_TO]->(to)
      // The instruction "createLine() — Line creation with endpoint validation" implies the Line itself is the connection.
      // Let's assume the Line *is* the relationship for this implementation, and `lineId` is a property *on* that relationship.
      // However, the test uses "Line {id: $lineId}" implying it's a node.
      // Re-reading: "Line deletion through Mutation Resonator" suggests Line is a node.
      // Let's create a Line node and connect it.

      const createLineAndRelationQuery = `
        MATCH (fromNode {id: $fromId}), (toNode {id: $toId})
        CREATE (line:Line $initialProperties)
        CREATE (fromNode)-[:${lineType} {createdAt: $timestamp, lineId: $lineId}]->(line)
        CREATE (line)-[:FLOWS_TO {createdAt: $timestamp, lineId: $lineId}]->(toNode)
        RETURN line.id AS lineId
      `;
      // This creates a Line NODE, and then creates two relationships:
      // (fromNode)-[LINE_TYPE]->(lineNode) AND (lineNode)-[:FLOWS_TO]->(toNode)
      // This allows the 'Line' to be directly identified by its 'lineId' property on the node itself.
      const lineCreateResult = await tx.run(createLineAndRelationQuery, { fromId, toId, lineType, lineId, initialProperties, timestamp });

      if (lineCreateResult.records.length === 0) {
        throw new Error("Failed to create line or link nodes.");
      }

      // 3. Record creation observation in grid:line-creation-observations
      const observationId = `obs-line-create:${lineId}:${Date.now()}`;
      const observationQuery = `
        CREATE (o:Observation:Grid:grid:line-creation-observations {
          id: $observationId,
          type: 'line_creation',
          lineId: $lineId,
          timestamp: $timestamp,
          outcome: 'success',
          message: 'Line created successfully'
        })
        RETURN o.id AS observationId
      `;
      await tx.run(observationQuery, { observationId, lineId, timestamp });

      return {
        success: true,
        lineId: lineCreateResult.records[0].get("lineId"),
      };
    });
    return result;
  } catch (error) {
    console.error(`Error creating line ${lineId}:`, error);
    return { success: false, error: `Failed to create line: ${error.message}` };
  }
}

/**
 * Marks a Line node for deletion and records an observation.
 * This function serves as the entry point for line deletion through the Mutation Resonator protocol.
 * The line is marked by setting its 'status' property to 'deleted' and adding a 'deletedAt' timestamp.
 *
 * @param lineId - The ID of the Line node to mark for deletion.
 * @returns A Result object indicating success or failure, with the ID of the deleted line if successful.
 */
export async function deleteLine(lineId: string): Promise<Result> {
  if (!lineId) {
    return { success: false, error: "lineId is required" };
  }

  const timestamp = new Date().toISOString();

  try {
    const result = await writeTransaction(async (tx) => {
      // 1. Mark the line for deletion (set status to 'deleted', add deletedAt timestamp)
      const lineDeleteQuery = `
        MATCH (l:Line {id: $lineId})
        SET l.status = 'deleted',
            l.deletedAt = $timestamp
        RETURN l.id AS lineId
      `;
      const lineQueryResult = await tx.run(lineDeleteQuery, { lineId, timestamp });

      if (lineQueryResult.records.length === 0) {
        // If the line was not found or not modified, it might not exist or already be deleted.
        return { success: false, error: `Line with ID ${lineId} not found or already deleted.` };
      }

      // 2. Record deletion observation in grid:line-creation-observations
      const observationId = `obs-line-del:${lineId}:${Date.now()}`; // Unique observation ID
      const observationQuery = `
        CREATE (o:Observation:Grid:grid:line-creation-observations {
          id: $observationId,
          type: 'line_deletion',
          lineId: $lineId,
          timestamp: $timestamp,
          outcome: 'success',
          message: 'Line marked as deleted'
        })
        RETURN o.id AS observationId
      `;
      await tx.run(observationQuery, { observationId, lineId, timestamp });

      return { success: true, lineId };
    });
    return result;
  } catch (error) {
    console.error(`Error deleting line ${lineId}:`, error);
    return { success: false, error: `Failed to delete line: ${error.message}` };
  }
}
