/**
 * Codex Signum -- Canonical Observation Writer with Inline Conditioning
 *
 * writeObservation() is the single entry point for recording observations
 * with inline health computation. There is no Observer, no collector, no
 * pipeline. These are function calls that happen as part of the write.
 *
 * Flow:
 *   1. Record raw Observation node (with rawValue preserved)
 *   2. conditionValue() -- 7-stage signal pipeline inline
 *   3. Persist conditioned values on the Observation node
 *   --- Steps 4-7 require PatternHealthContext (optional) ---
 *   4. computePhiL() -- recompute ΦL composite
 *   5. healthBand() -- classify into 6-band
 *   6. Detect band crossing --> CREATE immutable ThresholdEvent
 *   7. updatePatternPhiL() -- SET health on Pattern node
 *   8. Algedonic ΦL < 0.1 --> propagateDegradation() with dampening
 *
 * When context is omitted, steps 1-3 run (conditioning only).
 * When context is provided, the full chain runs (conditioning + ΦL + cascade).
 *
 * @module codex-signum-core/graph/write-observation
 */

/**

 */

  /** Number of retained observations (pre-increment; this write adds 1) */

}

// ============ CORE FUNCTIONS ============

/**
 * Record an observation with inline conditioning and optional ΦL recomputation.
 *
 * This is the single entry point for observation writes. Consumers should
 * use this instead of raw recordObservation().
 *
 * Signal conditioning (7-stage pipeline) ALWAYS runs — this is the M-22.1
 * vertical wiring. Conditioned values are persisted on the Observation node.
 *
 * ΦL recomputation, band classification, and cascade propagation only run
 * when PatternHealthContext is provided. Without context, the function
 * records + conditions the observation and returns.
 *
 * The SignalPipeline instance MUST be reused across calls to maintain
 * EWMA/CUSUM/trend accumulator state.
 *
 * @param observation - Raw observation properties (value = raw metric value)
 * @param pipeline - Signal pipeline instance (caller owns lifecycle, must reuse)
 * @param context - Pattern health context (optional — when provided, full chain runs)
 * @returns WriteObservationResult
 */
export async function writeObservation(
  observation: ObservationProps,
  pipeline: SignalPipeline,
  context?: PatternHealthContext,
): Promise<WriteObservationResult> {
  // Step 1: Record the raw observation in the graph
  await recordObservation(observation);

  // Step 2: Condition the value through the 7-stage signal pipeline (INLINE)
  const conditioned = conditionValue(
    pipeline,
    observation.sourceBloomId,
    observation.metric,
    observation.value,
    context?.topologyRole,
  );

  // Step 3: Persist conditioned values on the Observation node
  await updateObservationConditioned(observation.id, {
    smoothedValue: conditioned.smoothedValue,
    trendSlope: conditioned.trendSlope,
    trendProjection: conditioned.trendProjection,
    cusumStatistic: conditioned.cusumStatistic,
    macdValue: conditioned.macdValue,
    macdSignal: conditioned.macdSignal,
    filtered: conditioned.filtered,
    alertCount: conditioned.alerts.length,
  });

  // --- Without context, conditioning is all we do ---
  if (!context) {
    return {
      conditioned,
      phiL: null,
      band: null,
      thresholdEvent: null,
      cascadeResult: null,
    };
  }

  // Step 4: Recompute ΦL with updated observation count
  const phiL = computePhiL(
    context.factors,
    context.observationCount + 1, // This observation increments the count
    context.connectionCount,
    context.previousPhiL,
  );

  // Step 5: Classify health band (maturity-indexed, 6-band)
  const band = healthBand(phiL.effective, context.maturityIndex);

  // Step 6: Detect band crossing --> write immutable ThresholdEvent
  let thresholdEvent: ThresholdEvent | null = null;
  if (context.previousBand !== undefined && context.previousBand !== band) {
    thresholdEvent = await writeThresholdEvent(
      observation.sourceBloomId,
      context.previousBand,
      band,
      phiL.effective,
      context.maturityIndex,
    );
  }

  // Step 7: Update bloom's stored ΦL on the Bloom node
  // M-22.2: also persist healthBand (for cross-run band-crossing detection)
  // and updated PhiLState ring buffer (for temporal stability)
  let updatedPhiLStateJson: string | undefined;
  if (context.phiLState) {
    // Push new effective value into ring buffer (immutable update)
    const buf = [...context.phiLState.ringBuffer, phiL.effective];
    if (buf.length > context.phiLState.maxSize) buf.shift();
    const updatedState: PhiLState = { ...context.phiLState, ringBuffer: buf };
    updatedPhiLStateJson = JSON.stringify(updatedState);
  }

  await updateBloomPhiL(
    observation.sourceBloomId,
    phiL.effective,
    band,
    updatedPhiLStateJson,
  );

  // Step 8: Algedonic response — if ΦL fell below threshold, trigger cascade
  let cascadeResult: PropagationResult | null = null;
  if (phiL.effective < ALGEDONIC_THRESHOLD && context.neighbors) {
    cascadeResult = await propagateDegradation(
      observation.sourceBloomId,
      phiL.effective,
      context.neighbors,
      context.maturityIndex,
    );
  }

  // Upward propagation (M-22.3): notify parent of new ΦL and band
  // (parent recomputes its own ΦL via next observation)
  if (phiL.effective > PHI_L_PROPAGATION_NOISE_GATE) {
    // Only propagate significant changes to reduce graph churn
    await propagatePhiLUpward(observation.sourceBloomId, phiL.effective, band);
  }

  return {
    conditioned,
    phiL,
    band,
    thresholdEvent,
    cascadeResult,
  };
}

/**
 * Record a line deletion observation in grid:line-creation-observations.
 *
 * Called by deleteLine() when a Line (relationship) is removed from the graph.
 * Records the deletion event with outcome and context for queryable pattern
 * history and audit trail.
 *
 * Deletion observations are stored as Observation nodes linked to the
 * grid:line-creation-observations Grid, capturing:
 * - Which line was deleted (lineId)
 * - Source and target node identifiers
 * - Outcome (success/failed/queued)
 * - Timestamp of the deletion event
 *
 * The observation is recorded directly — no signal conditioning pipeline
 * applies (line deletions are discrete events, not continuous metrics).
 *
 * @param props - Line deletion observation details
 * @returns Result with success flag and observationId
 *
 * @example
 * ```typescript
 * const result = await writeLineDeletionObservation({
 *   id: uuid(),
 *   lineId: 'FLOWS_TO:node1->node2',
 *   lineSourceId: 'node1',
 *   lineTargetId: 'node2',
 *   outcome: 'success',
 *   timestamp: Date.now(),
 * });
 * ```
 */
export async function writeLineDeletionObservation(
  props: LineDeletionObservationProps,
): Promise<WriteDeletionObservationResult> {
  try {
    // Record the deletion observation in grid:line-creation-observations
    await recordGridObservation({
      id: props.id,
      gridId: "grid:line-creation-observations",
      eventType: "line-deletion",
      lineId: props.lineId,
      lineSourceId: props.lineSourceId,
      lineTargetId: props.lineTargetId,
      outcome: props.outcome,
      timestamp: props.timestamp,
      reason: props.reason,
    });

    return {
      success: true,
      observationId: props.id,
    };
  } catch (error) {
    const errorMessage =
      error instanceof Error ? error.message : "Unknown error";
    return {
      success: false,
      observationId: props.id,
      error: `Failed to record line deletion observation: ${errorMessage}`,
    };
  }
}

// ============ INTERNAL HELPERS ============

/**
 * Write an immutable ThresholdEvent when a band crossing is detected.
 *
 * ThresholdEvents are immutable records of ΦL state transitions.
 * They trigger downstream governance actions but do not carry observational
 * data themselves — they are pure events.
 *
 * @internal
 */
async function writeThresholdEvent(
  bloomId: string,
  fromBand: HealthBand,
  toBand: HealthBand,
  phiL: number,
  maturityIndex: number,
): Promise<ThresholdEvent> {
  const eventId = `threshold-event:${bloomId}:${Date.now()}`;
  const ordinal = bandOrdinal(toBand);

  const event: ThresholdEvent = {
    id: eventId,
    bloomId,
    fromBand,
    toBand,
    phiL,
    bandOrdinal: ordinal,
    maturityIndex,
    timestamp: Date.now(),
    immutable: true,
  };

  // Persist the event (implementation delegates to Neo4j via writeTransaction)
  await writeTransaction(async (tx) => {
    await tx.run(
      `
      CREATE (te:ThresholdEvent {
        id: $id,
        bloomId: $bloomId,
        fromBand: $fromBand,
        toBand: $toBand,
        phiL: $phiL,
        bandOrdinal: $ordinal,
        maturityIndex: $maturityIndex,
        timestamp: $timestamp,
        immutable: true
      })
      `,
      event,
    );
  });

  return event;
}

/**
 * Grid observation recording interface.
 *
 * Used to record discrete events (e.g., line creation/deletion) in a Grid's
 * observation history. Unlike Observation nodes tied to continuous metrics,
 * grid observations track mutations and lifecycle events.
 */
export interface GridObservationProps {
  /** Unique observation ID */
  id: string;
  /** Target grid ID (e.g., "grid:line-creation-observations") */
  gridId: string;
  /** Event type classifier (e.g., "line-creation", "line-deletion") */
  eventType: string;
  /** Line identifier (relationship ID or composite key) */
  lineId: string;
  /** Source node ID of the line */
  lineSourceId: string;
  /** Target node ID of the line */
  lineTargetId: string;
  /** Outcome: success | failed | queued */
  outcome: "success" | "failed" | "queued";
  /** Timestamp (milliseconds since epoch) */
  timestamp: number;
  /** Optional reason (error message, etc.) */
  reason?: string;
}

/**
 * Record a grid observation (discrete event) in a Grid's observation history.
 *
 * Grid observations differ from continuous Observation nodes — they capture
 * mutation events (line creation/deletion) as discrete records for audit and
 * pattern history.
 *
 * @param props - Grid observation details
 */
export async function recordGridObservation(
  props: GridObservationProps,
): Promise<void> {
  await writeTransaction(async (tx) => {
    // Create the observation node
    await tx.run(
      `
      CREATE (go:Observation {
        id: $id,
        gridId: $gridId,
        eventType: $eventType,
        lineId: $lineId,
        lineSourceId: $lineSourceId,
        lineTargetId: $lineTargetId,
        outcome: $outcome,
        timestamp: $timestamp,
        reason: $reason,
        metric: 'line-mutation',
        value: 0,
        filtered: true,
        recordedAt: timestamp()
      })
      `,
      {
        id: props.id,
        gridId: props.gridId,
        eventType: props.eventType,
        lineId: props.lineId,
        lineSourceId: props.lineSourceId,
        lineTargetId: props.lineTargetId,
        outcome: props.outcome,
        timestamp: props.timestamp,
        reason: props.reason || null,
      },
    );

    // Link observation to the Grid (if grid exists)
    await tx.run(
      `
      MATCH (g:Grid {id: $gridId})
      MATCH (go:Observation {id: $obsId})
      CREATE (go)-[:OBSERVED_IN]->(g)
      `,
      {
        gridId: props.gridId,
        obsId: props.id,
      },
    );
  });
}
