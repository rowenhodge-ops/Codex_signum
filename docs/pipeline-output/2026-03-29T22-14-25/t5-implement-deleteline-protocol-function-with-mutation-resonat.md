/**
 * Create a morpheme instance via the Instantiation Resonator.
 *
 * Atomic transaction: node creation + CONTAINS wiring + INSTANTIATES wiring.
 * Records observation in the Instantiation Resonator's Grid.
 *
 * @param morphemeType - One of: seed, bloom, resonator, grid, helix
 * @param properties - All properties for the node (must include required fields)
 * @param parentId - The Bloom (or Grid for seeds) that will CONTAIN this morpheme
 */
export async function instantiateMorpheme(
  morphemeType: MorphemeType,
  properties: Record<string, unknown>,
  parentId: string,
  highlander?: HighlanderOptions,
  options?: InstantiationOptions,
): Promise<InstantiationResult> {
  const nodeId = properties.id as string | undefined;

  // ── Step 1: Morpheme hygiene check ──
  const required = REQUIRED_PROPERTIES[morphemeType];
  for (const prop of required) {
    const val = properties[prop];
    if (val === undefined || val === null || val === "") {
      const error = `Instantiation rejected: ${morphemeType} missing required property '${prop}'. id=${nodeId ?? "unknown"}`;
      await recordInstantiationObservation(morphemeType, nodeId ?? "unknown", parentId, false, error);
      return { success: false, error };
    }
  }

  // Content must be non-empty (applies to ALL morpheme types)
  const content = properties.content as string;
  if (!content || content.trim() === "") {
    const error = `Instantiation rejected: ${morphemeType} '${nodeId}' has empty content. Every morpheme carries meaning (A1).`;
    await recordInstantiationObservation(morphemeType, nodeId ?? "unknown", parentId, false, error);
    return { success: false, error };
  }

  // ── Step 2: Grammatical shape check ──
  // Verify the parent can contain this morpheme type
  const parentLabel = await getNodeLabel(parentId);
  if (!parentLabel) {
    const error = `Instantiation rejected: parent '${parentId}' not found`;
    await recordInstantiationObservation(morphemeType, nodeId ?? "unknown", parentId, false, error);
    return { success: false, error };
  }

  const parentType = parentLabel.toLowerCase() as MorphemeType;
  const allowedChildren = VALID_CONTAINERS[parentType];
  if (!allowedChildren || !allowedChildren.includes(morphemeType)) {
    const error = `Instantiation rejected: ${parentType} cannot CONTAIN ${morphemeType}`;
    await recordInstantiationObservation(morphemeType, nodeId ?? "unknown", parentId, false, error);
    return { success: false, error };
  }

  // ── Step 3: Highlander Protocol (resonator/bloom/grid/helix only) ──
  if (["resonator", "bloom", "grid", "helix"].includes(morphemeType)) {
    if (!highlander?.transformationDefId) {
      const error = `Instantiation rejected: ${morphemeType} requires transformationDefId (Highlander Protocol)`;
      await recordInstantiationObservation(morphemeType, nodeId ?? "unknown", parentId, false, error);
      return { success: false, error };
    }

    // Check if an instance already exists for this transformation
    const existingInstance = await findExistingInstance(highlander.transformationDefId);

    if (existingInstance) {
      // Instance exists — check for A6 justification
      if (!highlander.a6Justification) {
        // No justification → COMPOSE instead of creating new instance
        if (!highlander.requestingContextId) {
          const error = `Instantiation rejected: existing instance of '${highlander.transformationDefId}' found, but no requestingContextId for composition`;
          await recordInstantiationObservation(morphemeType, nodeId ?? "unknown", parentId, false, error);
          return { success: false, error };
        }

        // Wire FLOWS_TO from requesting context to existing instance
        const lineResult = await createLine(
          highlander.requestingContextId,
          existingInstance.id,
          "FLOWS_TO",
        );

        return {
          success: true,
          composed: {
            composed: true,
            existingId: existingInstance.id,
            existingName: existingInstance.name,
            lineCreated: lineResult.success,
            lineError: lineResult.error,
          },
        };
      }

      // A6 justification provided — validate it
      if (!VALID_A6_JUSTIFICATIONS.includes(highlander.a6Justification)) {
        const error = `Instantiation rejected: invalid A6 justification '${highlander.a6Justification}'`;
        await recordInstantiationObservation(morphemeType, nodeId ?? "unknown", parentId, false, error);
        return { success: false, error };
      }

      // Valid justification → proceed with new instance creation
      // (justification is recorded as a property below)
    }
  }

  // ── Step 4: Execute instantiation transaction ──
  try {
    const label = LABEL_MAP[morphemeType];
    const defId = DEFINITION_MAP[morphemeType];

    // Add secondary label for seed sub-types (Option B multi-label retyping)
    const labels = [label];
    if (morphemeType === 'seed' && options?.subType) {
      if (!VALID_SEED_SUBTYPES.includes(options.subType)) {
        const error = `Instantiation rejected: invalid seed subType '${options.subType}'`;
        await recordInstantiationObservation(morphemeType, nodeId ?? "unknown", parentId, false, error);
        return { success: false, error };
      }
      labels.push(options.subType);
    }

    const result = await writeTransaction(async (tx) => {
      // Serialize properties for Cypher
      const propEntries = Object.entries(properties).map(
        ([key, val]) => `${key}: $${key}`,
      );

      // Add A6 justification if present
      if (highlander?.a6Justification) {
        propEntries.push("a6Justification: $a6Justification");
      }

      const propsString = `{${propEntries.join(", ")}}`;
      const labelString = labels.join(":");

      const query = `
        MATCH (parent {id: $parentId})
        MATCH (def:Seed {id: $defId})
        CREATE (n:${labelString} ${propsString})
        CREATE (parent)-[:CONTAINS]->(n)
        CREATE (n)-[:INSTANTIATES]->(def)
        RETURN n.id AS nodeId
      `;

      const params: Record<string, unknown> = {
        ...properties,
        parentId,
        defId,
      };

      if (highlander?.a6Justification) {
        params.a6Justification = highlander.a6Justification;
      }

      const txResult = await tx.run(query, params);
      return txResult.records[0]?.get("nodeId") as string;
    });

    if (!result) {
      throw new Error("Transaction returned no nodeId");
    }

    // ── Step 5: Record successful instantiation ──
    await recordInstantiationObservation(morphemeType, result, parentId, true, undefined);

    return { success: true, nodeId: result };
  } catch (error) {
    const errorMsg = error instanceof Error ? error.message : String(error);
    await recordInstantiationObservation(morphemeType, nodeId ?? "unknown", parentId, false, errorMsg);
    return { success: false, error: errorMsg };
  }
}

/**
 * Update a morpheme instance via the Mutation Resonator.
 *
 * Enforces constitutional hygiene: properties are validated, the change is recorded,
 * and the result is verified by reading back the updated node.
 *
 * @param nodeId - The morpheme instance to update
 * @param updates - Properties to update (partial — only changed fields)
 * @param addLabels - Optional labels to add (allowlist: ['Archived']) — M-10.1 requirement 7
 */
export async function updateMorpheme(
  nodeId: string,
  updates: Record<string, unknown>,
  addLabels?: string[],
): Promise<MutationResult> {
  // ── Step 1: Hygiene ──
  if (!nodeId || nodeId.trim() === "") {
    const error = "Mutation rejected: nodeId cannot be empty";
    await recordMutationObservation(nodeId, false, error);
    return { success: false, error };
  }

  if (Object.keys(updates).length === 0 && (!addLabels || addLabels.length === 0)) {
    const error = "Mutation rejected: no updates or labels provided";
    await recordMutationObservation(nodeId, false, error);
    return { success: false, error };
  }

  // Validate addLabels allowlist (M-10.1 requirement 7)
  const ALLOWED_ADD_LABELS = ['Archived'];
  if (addLabels) {
    for (const label of addLabels) {
      if (!ALLOWED_ADD_LABELS.includes(label)) {
        const error = `Mutation rejected: label '${label}' not in allowlist ${JSON.stringify(ALLOWED_ADD_LABELS)}`;
        await recordMutationObservation(nodeId, false, error);
        return { success: false, error };
      }
    }
  }

  // ── Step 2: Execute mutation ──
  try {
    const result = await writeTransaction(async (tx) => {
      // Build SET clause for property updates
      const setClauses: string[] = [];
      if (Object.keys(updates).length > 0) {
        const setProps = Object.keys(updates)
          .map((key) => `n.${key} = $${key}`)
          .join(", ");
        setClauses.push(setProps);
      }

      // Build label addition clause
      let labelClause = "";
      if (addLabels && addLabels.length > 0) {
        labelClause = `SET n:${addLabels.join(":")}`;
      }

      const query = `
        MATCH (n {id: $nodeId})
        ${setClauses.length > 0 ? `SET ${setClauses.join(", ")}` : ""}
        ${labelClause}
        RETURN n.id AS nodeId
      `;

      const params: Record<string, unknown> = {
        nodeId,
        ...updates,
      };

      const txResult = await tx.run(query, params);
      return txResult.records[0]?.get("nodeId") as string;
    });

    if (!result) {
      throw new Error(`Node '${nodeId}' not found`);
    }

    // ── Step 3: Verify mutation persisted ──
    const verifyQuery = `MATCH (n {id: $nodeId}) RETURN n`;
    const verifyResult = await runQuery(verifyQuery, { nodeId });

    if (verifyResult.records.length === 0) {
      throw new Error("Verification failed: node not found after mutation");
    }

    const node = verifyResult.records[0].get("n");
    for (const [key, expectedValue] of Object.entries(updates)) {
      const actualValue = node.properties[key];
      if (actualValue !== expectedValue) {
        throw new Error(
          `Verification failed: property '${key}' is ${actualValue}, expected ${expectedValue}`,
        );
      }
    }

    // ── Step 4: Record successful mutation ──
    await recordMutationObservation(nodeId, true, undefined);

    return { success: true, verified: true, nodeId: result };
  } catch (error) {
    const errorMsg = error instanceof Error ? error.message : String(error);
    await recordMutationObservation(nodeId, false, errorMsg);
    return { success: false, error: errorMsg };
  }
}

/**
 * Create a Line (relationship) via the Line Creation Resonator.
 *
 * Enforces Line type validity and records the creation event in the
 * Line Creation Resonator's Grid.
 *
 * @param sourceId - Source morpheme
 * @param targetId - Target morpheme
 * @param lineType - Relationship type (must be in VALID_LINE_TYPES)
 */
export async function createLine(
  sourceId: string,
  targetId: string,
  lineType: string,
): Promise<LineCreationResult> {
  // ── Step 1: Validate line type ──
  if (!VALID_LINE_TYPES.includes(lineType as LineType)) {
    const error = `Line creation rejected: invalid lineType '${lineType}'`;
    await recordLineCreationObservation(sourceId, targetId, lineType, false, error);
    return { success: false, error };
  }

  // ── Step 2: Execute line creation ──
  try {
    await writeTransaction(async (tx) => {
      const query = `
        MATCH (source {id: $sourceId})
        MATCH (target {id: $targetId})
        CREATE (source)-[:${lineType}]->(target)
      `;

      await tx.run(query, { sourceId, targetId });
    });

    // ── Step 3: Record successful creation ──
    await recordLineCreationObservation(sourceId, targetId, lineType, true, undefined);

    // ── Step 4: Invalidate conductivity cache if this is a FLOWS_TO line ──
    if (lineType === "FLOWS_TO") {
      await invalidateLineConductivity(sourceId, targetId);
      // Optionally recompute immediately
      await evaluateAndCacheLineConductivity(sourceId, targetId);
    }

    return { success: true, sourceId, targetId, lineType };
  } catch (error) {
    const errorMsg = error instanceof Error ? error.message : String(error);
    await recordLineCreationObservation(sourceId, targetId, lineType, false, errorMsg);
    return { success: false, error: errorMsg };
  }
}

/**
 * Delete a Line (relationship) via the Mutation Resonator.
 *
 * Routes line deletion through the Mutation Resonator pattern,
 * recording the deletion event in grid:line-creation-observations.
 * Handles missing lines gracefully.
 *
 * @param lineId - Composite identifier for the line: "sourceId|lineType|targetId"
 * @returns LineDeletionResult indicating success/failure
 *
 * @see M-10.1 requirement 6
 * @see R-61
 */
export async function deleteLine(lineId: string): Promise<LineDeletionResult> {
  // ── Step 1: Hygiene ──
  if (!lineId || lineId.trim() === "") {
    const error = "Line deletion rejected: lineId cannot be empty";
    await recordLineCreationObservation("unknown", "unknown", "unknown", false, error);
    return { success: false, error };
  }

  // Parse composite lineId: "sourceId|lineType|targetId"
  const parts = lineId.split("|");
  if (parts.length !== 3) {
    const error = `Line deletion rejected: lineId must be in format 'sourceId|lineType|targetId', got '${lineId}'`;
    await recordLineCreationObservation("unknown", "unknown", "unknown", false, error);
    return { success: false, error };
  }

  const [sourceId, lineType, targetId] = parts;

  // Validate line type
  if (!VALID_LINE_TYPES.includes(lineType as LineType)) {
    const error = `Line deletion rejected: invalid lineType '${lineType}'`;
    await recordLineCreationObservation(sourceId, targetId, lineType, false, error);
    return { success: false, error };
  }

  // ── Step 2: Check if line exists ──
  try {
    const checkQuery = `
      MATCH (source {id: $sourceId})-[r:${lineType}]->(target {id: $targetId})
      RETURN count(r) > 0 AS exists
    `;

    const checkResult = await runQuery(checkQuery, { sourceId, targetId });
    const exists = checkResult.records[0]?.get("exists");

    if (!exists) {
      // Line not found — handle gracefully (R-61)
      const error = `Line '${lineId}' not found`;
      await recordLineCreationObservation(sourceId, targetId, lineType, false, error);
      return { success: false, error };
    }

    // ── Step 3: Execute deletion ──
    await writeTransaction(async (tx) => {
      const query = `
        MATCH (source {id: $sourceId})-[r:${lineType}]->(target {id: $targetId})
        DELETE r
      `;

      await tx.run(query, { sourceId, targetId });
    });

    // ── Step 4: Record successful deletion ──
    await recordLineCreationObservation(sourceId, targetId, lineType, true, undefined);

    // ── Step 5: Invalidate conductivity cache if this was a FLOWS_TO line ──
    if (lineType === "FLOWS_TO") {
      await invalidateLineConductivity(sourceId, targetId);
    }

    return { success: true, sourceId, targetId, lineType };
  } catch (error) {
    const errorMsg = error instanceof Error ? error.message : String(error);
    await recordLineCreationObservation(sourceId, targetId, lineType, false, `Deletion error: ${errorMsg}`);
    return { success: false, error: errorMsg };
  }
}

/**
 * Stamp a Bloom as complete with derived ΦL (phiL).
 *
 * This is the ONLY function that writes `phiL` and `stampedAt` to a Bloom.
 * Triggers hierarchical ΦL propagation and immune response evaluation.
 *
 * @param bloomId - The Bloom to stamp
 */
export async function stampBloomComplete(bloomId: string): Promise<MutationResult> {
  try {
    // ── Step 1: Verify not already stamped ──
    const stampCheck = await runQuery(
      `MATCH (b:Bloom {id: $bloomId}) RETURN b.stampedAt AS stampedAt`,
      { bloomId },
    );

    if (stampCheck.records.length === 0) {
      throw new Error(`Bloom '${bloomId}' not found`);
    }

    const alreadyStamped = stampCheck.records[0].get("stampedAt");
    if (alreadyStamped) {
      return {
        success: false,
        error: `Bloom '${bloomId}' already stamped at ${alreadyStamped}`,
      };
    }

    // ── Step 2: Derive ΦL from child observations ──
    // (This will be replaced by M-10.1 requirement 4 — reading from γ-recursive properties)
    const phiL = await derivePhiL(bloomId);

    // ── Step 3: Write stamp atomically ──
    const stampedAt = Date.now();
    await writeTransaction(async (tx) => {
      const query = `
        MATCH (b:Bloom {id: $bloomId})
        SET b.phiL = $phiL, b.stampedAt = $stampedAt, b.status = 'complete'
        RETURN b.id
      `;

      await tx.run(query, { bloomId, phiL, stampedAt });
    });

    // ── Step 4: Propagate ΦL upward to parent Blooms ──
    const parentBloom = await getParentBloom(bloomId);
    if (parentBloom) {
      await propagatePhiLUpward(parentBloom);
    }

    // ── Step 5: Recompute Ψ_H and ε_R with fresh ΦL ──
    await computeAndPersistPsiH();
    await computeAndPersistEpsilonR();

    // ── Step 6: Evaluate immune triggers ──
    const triggerState = await assembleTriggerState();
    // (Trigger evaluation logic continues in immune-response.ts)

    return { success: true, verified: true, nodeId: bloomId };
  } catch (error) {
    const errorMsg = error instanceof Error ? error.message : String(error);
    return { success: false, error: errorMsg };
  }
}

/**
 * Record an instantiation attempt in the Instantiation Resonator's Grid.
 */
async function recordInstantiationObservation(
  morphemeType: string,
  morphemeId: string,
  parentId: string,
  success: boolean,
  errorMessage: string | undefined,
): Promise<void> {

    const timestamp = Date.now();
    const observationId = `obs:instantiation:${morphemeId}:${timestamp}`;

    await runQuery(
      `
      MATCH (grid:Grid {id: 'grid:instantiation-observations'})

/**
 * Get the primary label of a node (first label returned by Neo4j).
 */
async function getNodeLabel(nodeId: string): Promise<string | null> {
  const result = await runQuery(
    `MATCH (n {id: $nodeId}) RETURN labels(n)[0] AS label`,
    { nodeId },
  );

  if (result.records.length === 0) {
    return null;
  }

  return result.records[0].get("label");
}

/**
 * Find an existing instance of a transformation definition.
 * Returns the first instance found (there should be only one unless A6-justified).
 */
async function findExistingInstance(
  transformationDefId: string,
): Promise<{ id: string; name: string } | null> {
  const result = await runQuery(
    `
    MATCH (def:Seed {id: $transformationDefId})<-[:INSTANTIATES]-(instance)
    RETURN instance.id AS id, instance.name AS name
    LIMIT 1
  `,
    { transformationDefId },
  );

  if (result.records.length === 0) {
    return null;
  }

  const record = result.records[0];
  return {
    id: record.get("id"),
    name: record.get("name"),
  };
}
